Code Examples

How to use Terminal Toggles in your application.

<!-- Toggle switches --> <t-tog label="Enable feature"></t-tog> <t-tog label="Auto-save" checked></t-tog> <!-- Icon toggles --> <t-tog icon-on="♥" icon-off="♡"></t-tog> <t-tog icon-on="●" icon-off="○" label="Notifications"></t-tog> <!-- Switching toggles --> <t-tog label-on="ON" label-off="OFF" label="Status"></t-tog> <t-tog icon-on="+" icon-off="-"></t-tog> <!-- Equal-state toggles --> <t-tog label-on="Mode A" label-off="Mode B" equal-states label="Mode"></t-tog> <!-- Sizes --> <t-tog label="Small" size="small"></t-tog> <t-tog label="Large" size="large"></t-tog> <!-- States --> <t-tog label="Checked" checked></t-tog> <t-tog label="Disabled" disabled></t-tog> <!-- Checkbox variant --> <t-tog variant="checkbox" label="Option 1"></t-tog> <t-tog variant="checkbox" label="Option 2" checked></t-tog> <t-tog variant="checkbox" label="Option 3" disabled></t-tog> <!-- Colored checkboxes --> <t-tog variant="checkbox" label="Error" color-scheme="error" checked></t-tog> <t-tog variant="checkbox" label="Warning" color-scheme="warning" checked></t-tog> <t-tog variant="checkbox" label="Success" color-scheme="success" checked></t-tog> <!-- Alignment options --> <t-tog variant="checkbox" label="Left aligned" alignment="left"></t-tog> <t-tog variant="checkbox" label="Right aligned" alignment="right"></t-tog>
// JavaScript API const toggle = document.querySelector('t-tog'); // Get/set checked state console.log(toggle.checked); toggle.checked = true; // Set color scheme toggle.colorScheme = 'error'; // or 'warning', 'success' // Set alignment toggle.alignment = 'left'; // or 'right' // Enable/disable toggle.disabled = true; toggle.disabled = false; // Event listeners toggle.addEventListener('change', (e) => { console.log('Toggle changed:', e.target.checked); }); // Setting icons programmatically toggle.setAttribute('icon', iconSVG); toggle.setAttribute('icon-on', onIconSVG); toggle.setAttribute('icon-off', offIconSVG);
