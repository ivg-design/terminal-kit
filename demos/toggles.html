Code Examples

How to use Terminal Toggles in your application.

<!-- Toggle switches --> <terminal-toggle label="Enable feature"></terminal-toggle> <terminal-toggle label="Auto-save" checked></terminal-toggle> <!-- Icon toggles --> <terminal-toggle layout="icon-toggle"></terminal-toggle> <terminal-toggle layout="icon-label-toggle" label="Notifications"></terminal-toggle> <!-- Switching toggles --> <terminal-toggle layout="switching" on-label="ONLINE" off-label="OFFLINE"></terminal-toggle> <terminal-toggle layout="icon-switching"></terminal-toggle> <!-- Equal-state toggles --> <terminal-toggle layout="switching" equal-states on-label="WORK" off-label="PERSONAL"></terminal-toggle> <!-- Sizes --> <terminal-toggle label="Small" size="small"></terminal-toggle> <terminal-toggle label="Large" size="large"></terminal-toggle> <!-- States --> <terminal-toggle label="Checked" checked></terminal-toggle> <terminal-toggle label="Disabled" disabled></terminal-toggle> <!-- Checkbox variant --> <terminal-toggle variant="checkbox" label="Option 1"></terminal-toggle> <terminal-toggle variant="checkbox" label="Option 2" checked></terminal-toggle> <terminal-toggle variant="checkbox" label="Option 3" disabled></terminal-toggle> <!-- Error checkbox (shows X instead of checkmark) --> <terminal-toggle variant="checkbox" error label="Invalid" checked></terminal-toggle>
// JavaScript API const toggle = document.querySelector('terminal-toggle'); // Get/set checked state console.log(toggle.checked); toggle.checked = true; // Enable/disable toggle.disabled = true; toggle.disabled = false; // Event listeners toggle.addEventListener('change', (e) => { console.log('Toggle changed:', e.target.checked); }); // Setting icons programmatically toggle.setAttribute('icon', iconSVG); toggle.setAttribute('on-icon', onIconSVG); toggle.setAttribute('off-icon', offIconSVG);
