Code Examples

How to use the Terminal Toast Manager in your application. 

<!-- Toast HTML structure is automatically managed -->
<!-- Just import the component and use the ToastManager API -->

<script type="module">
import './js/components/TerminalToast.js';

// Simple toast notifications
window.TerminalToastManager.success('Operation completed successfully!');
window.TerminalToastManager.error('Something went wrong');
window.TerminalToastManager.warning('Please review your settings');
window.TerminalToastManager.info('New update available');

// Custom toast with options
window.TerminalToastManager.show('Custom message', 'info', {
  duration: 5000,        // Auto-dismiss after 5 seconds (0 = no auto-dismiss)
  position: 'top-right', // Position on screen
  dismissible: true,     // Show close button
  actions: [             // Custom action buttons
    {
      text: 'Undo',
      callback: () => console.log('Undo clicked')
    }
  ]
});
</script>
// Advanced Toast Manager API
const toastManager = window.TerminalToastManager;

// Queue management
toastManager.dismissAll();        // Clear all toasts
toastManager.getActiveCount();    // Get number of active toasts

// Toast with custom styling
toastManager.show('Custom toast', 'success', {
  duration: 3000,
  position: 'bottom-center',
  className: 'custom-toast-class'
});

// Toast with progress indicator
let progress = 0;
const progressToast = toastManager.show('Processing...', 'info', {
  duration: 0,  // Don't auto-dismiss
  progress: progress
});

const interval = setInterval(() => {
  progress += 10;
  progressToast.setProgress(progress);
  
  if (progress >= 100) {
    clearInterval(interval);
    progressToast.dismiss();
    toastManager.success('Process completed!');
  }
}, 200);

// Listen for toast events
document.addEventListener('toast-shown', (e) => {
  console.log('Toast shown:', e.detail);
});

document.addEventListener('toast-dismissed', (e) => {
  console.log('Toast dismissed:', e.detail);
});

document.addEventListener('toast-action', (e) => {
  console.log('Toast action clicked:', e.detail);
});
