Component API

Properties

message: string     // Toast message text
type: string        // 'success' | 'error' | 'warning' | 'info'
duration: number    // Auto-dismiss time in ms (0 = manual)
visible: boolean    // Whether toast is visible
position: string    // 'top-right' | 'top-left' | 'bottom-right' | 'bottom-left'

Methods

show()     // Show the toast
dismiss()  // Hide the toast

Events

toast-show     // Fired when toast is shown
toast-hide     // Fired when toast is hidden
toast-click    // Fired when toast is clicked
toast-dismiss  // Fired when toast is dismissed
