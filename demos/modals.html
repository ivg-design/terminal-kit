Code Examples

How to use Terminal Modals in your application. 

<!-- Basic modal with different sizes -->
<terminal-modal id="my-modal" size="md" title="Modal Title">
    <p>Modal content goes here...</p>
    <div style="margin-top: 20px;">
        <terminal-button variant="primary">Action</terminal-button>
        <terminal-button variant="secondary">Cancel</terminal-button>
    </div>
</terminal-modal>

<!-- Different sizes: sm, md, lg, xl, full -->
<terminal-modal size="sm" title="Small">...</terminal-modal>
<terminal-modal size="lg" title="Large">...</terminal-modal>
<terminal-modal size="full" title="Full Screen">...</terminal-modal>

<!-- Different layouts -->
<terminal-modal layout="single" title="Single Column">...</terminal-modal>
<terminal-modal layout="2-column" title="Two Columns">...</terminal-modal>
<terminal-modal layout="2x2" title="Grid Layout">...</terminal-modal>
// JavaScript API
const modal = document.querySelector('#my-modal');

// Show/hide modal
modal.show();
modal.hide();

// Check if modal is open
if (modal.isOpen()) {
    // Modal is currently visible
}

// Listen to modal events
modal.addEventListener('modal-opened', () => {
    console.log('Modal opened');
});

modal.addEventListener('modal-closed', () => {
    console.log('Modal closed');
});

// Helper functions
function showModal(modalId) {
    document.getElementById(modalId).show();
}

function hideModal(modalId) {
    document.getElementById(modalId).hide();
}
