Code Examples

How to use Terminal User Menu in your application. 

<!-- Basic user menu -->
<t-usr user-name="John Doe"></t-usr>

<!-- With avatar -->
<t-usr
  user-name="Jane Smith"
  user-avatar="/path/to/avatar.jpg">
</t-usr>

<!-- With email -->
<t-usr
  user-name="Dev User"
  user-email="dev@example.com"
  user-avatar="https://i.pravatar.cc/150?img=3">
</t-usr>

<!-- Disabled state -->
<t-usr user-name="Disabled" disabled></t-usr>
// JavaScript API
const userMenu = document.querySelector('t-usr');

// Get user info
console.log(userMenu.userName);
console.log(userMenu.userEmail);
console.log(userMenu.userAvatar);

// Update user info
userMenu.userName = 'New Name';
userMenu.userEmail = 'newemail@example.com';
userMenu.userAvatar = '/path/to/new/avatar.jpg';

// Update menu items
userMenu.setMenuItems([
    { id: 'custom', label: 'Custom Item', icon: 'gear' },
    { type: 'separator' },
    { id: 'signout', label: 'Sign Out', icon: 'signOut' }
]);

// Open/close/toggle menu
userMenu.openMenu();
userMenu.closeMenu();
userMenu.toggleMenu();

// Enable/disable
userMenu.disabled = true;
userMenu.disabled = false;

// Event listeners
userMenu.addEventListener('menu-open', () => {
    console.log('User menu opened');
});

userMenu.addEventListener('menu-close', () => {
    console.log('User menu closed');
});

userMenu.addEventListener('menu-select', (e) => {
    console.log('Menu item selected:', e.detail.itemId);
});
