Code Examples

How to use Terminal User Menu in your application. 

<!-- Basic user menu -->
<terminal-user-menu user-name="John Doe"></terminal-user-menu>

<!-- With avatar -->
<terminal-user-menu 
  user-name="Jane Smith"
  user-avatar="/path/to/avatar.jpg">
</terminal-user-menu>

<!-- With email -->
<terminal-user-menu 
  user-name="Dev User"
  user-email="dev@example.com"
  user-avatar="https://i.pravatar.cc/150?img=3">
</terminal-user-menu>

<!-- Disabled state -->
<terminal-user-menu user-name="Disabled" disabled></terminal-user-menu>
// JavaScript API
const userMenu = document.querySelector('terminal-user-menu');

// Get user info
console.log(userMenu.userName);
console.log(userMenu.userEmail);
console.log(userMenu.userAvatar);

// Update user info
userMenu.userName = 'New Name';
userMenu.userEmail = 'newemail@example.com';
userMenu.userAvatar = '/path/to/new/avatar.jpg';

// Enable/disable
userMenu.disabled = true;
userMenu.disabled = false;

// Event listeners
userMenu.addEventListener('menu-open', () => {
    console.log('User menu opened');
});

userMenu.addEventListener('menu-close', () => {
    console.log('User menu closed');
});

userMenu.addEventListener('menu-item-click', (e) => {
    console.log('Menu item clicked:', e.detail.item);
});
